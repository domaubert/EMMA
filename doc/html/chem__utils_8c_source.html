119 
120 
121  // Bremmsstrahlung
122 
123  c4=1.42e-27*1.5e0*SQRT(temp)*x*x*nh2*nh2*CLUMPF;

134 #endif

137  *lambda=c1+c2+c3+c4+c5+c6;// ! erg*cm-3*s-1
138 
139 
140  // Unit Conversion
141

144  // cooling times
145

147  unsurtc=FMAX(unsurtc,c3);
148  unsurtc=FMAX(unsurtc,c4);
149  unsurtc=FMAX(unsurtc,FABS(c5));
150  unsurtc=FMAX(unsurtc,c6)*1e-7;// ==> J/cm3/s

153 }
