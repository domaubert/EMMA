65 void kineticFeedback(struct RUNPARAMS *param, struct CELL *cell,struct PART *curp, REAL aexp, int level, REAL E){
66 // ----------------------------------------------------------//
72 // ----------------------------------------------------------//
73 
74  REAL ejecta_proportion = 0.5260172663907063;
75  REAL mtot_feedback = curp->mass* ejecta_proportion;
76 
77  int i;
78  for(i=0;i<8;i++){
79  struct OCT* oct = cell2oct(cell);
80  struct CELL* curcell = &oct->cell[i];
81 
82  REAL e = E/8.; //uniform energy distribution
83  REAL me = mtot_feedback/8.; //uniform ejecta distribution
84 
85  REAL dv = POW(2.,-3*level);// curent volume
86 
87  REAL rho_i = curcell->field.d; //initial density
88  REAL rho_e = me/dv; // density ejecta
89 
90  REAL vxi = curcell->field.u; // initial velocity
91  REAL vyi = curcell->field.v;
92  REAL vzi = curcell->field.w;
93 
94  REAL ve = SQRT(2.*e/rho_e);// ejecta velocity
95 
96  REAL dir_x[]={-1., 1.,-1., 1.,-1., 1.,-1., 1.};// diagonal projection
97  REAL dir_y[]={-1.,-1., 1., 1.,-1.,-1., 1., 1.};
98  REAL dir_z[]={-1.,-1.,-1.,-1., 1., 1., 1., 1.};
99 
100  REAL vxe = curp->vx + ve*dir_x[i]/2.; // projection on axis in the particle framework
101  REAL vye = curp->vy + ve*dir_y[i]/2.; // cos45*cos45 = 1/2
102  REAL vze = curp->vz + ve*dir_z[i]/2.;
103 
104  curcell->field.d += rho_e; //new density
105  curp->mass -= me; // new particle mass
106 
107  curcell->field.u = (vxi*rho_i + vxe*rho_e)/(rho_i+rho_e); //new velocity
108  curcell->field.v = (vyi*rho_i + vye*rho_e)/(rho_i+rho_e);
109  curcell->field.w = (vzi*rho_i + vze*rho_e)/(rho_i+rho_e);
110 
111  //Energy conservation
112 
113  struct Utype U; // conservative field structure
114  W2U(&curcell->field, &U); // primitive to conservative
115  U.eint*=1.+rho_e/curcell->field.d; // compute new internal energy
116  U2W(&U, &curcell->field); // back to primitive
117 
118  getE(&curcell->field); //compute new total energy
119  curcell->field.a=SQRT(GAMMA*curcell->field.p/curcell->field.d); // compute new sound speed
120  }
121 }
