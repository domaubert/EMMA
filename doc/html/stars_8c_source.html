102 void conserveField(struct Wtype *field, struct RUNPARAMS *param, struct PART *star, REAL dx, REAL aexp, REAL mlevel){
103 // ----------------------------------------------------------//
105 // ----------------------------------------------------------//
106 
107  REAL drho = mlevel / POW(dx,3.);
108  struct Utype U;
109  struct Wtype W;
110 
111  memcpy(&W,field,sizeof(struct Wtype));
112 
113  W2U(&W, &U);
114 
115 // density
116  U.d -= drho;
117 
118 #ifdef WRADHYD
119  REAL xion=W.dX/W.d;
120  U.dX = U.d*xion;
121 #endif
122 
123 // momentum
124  U.du -= star->vx * drho;
125  U.dv -= star->vy * drho;
126  U.dw -= star->vz * drho;
127 
128 // internal energy
129  U.eint=U.eint*(1.-drho/W.d); // assuming T and x remain constant
130 
131  U2W(&U, &W);
132 
133 // total energy
134  getE(&(W));
135  W.a=SQRT(GAMMA*W.p/W.d);
136  memcpy(field,&W,sizeof(struct Wtype));

152  REAL E=1.; // Ms/pc2

159  //printf("A=%e E=%e P=%e p=%e c=%e tstars=%e\n",A,E,P,cell->field.p,param->unit.unit_d,tstars/(3600.*24*265*1e9));

165 
166  REAL lambda = param->stars->efficiency * M_in_cell / mlevel * dt/ tstartilde; // Average number of stars created
167 #else
168  // local free fall time in seconde in code unit
169  // REAL t_ff = 1. / SQRT(6*aexp*cell->gdata.d);
170  //printf("Local SFR=%e M0/yr/Mpc3\n", SFR/SOLAR_MASS*31556926*POW(PARSEC,3));
171 
172  REAL dx = POW(0.5,level);
173  REAL dv = POW(0.5,3*level);
174 
175  REAL rho_m = cell->gdata.d / param->stars->thresh;
176  REAL rho_b = cell->field.d / param->stars->thresh;
177 
178  REAL fact_rho = POW(aexp,3)/param->unit.unit_d;
179  REAL fact_t = POW(aexp,2) * param->unit.unit_t;
180 
181  // local free fall time in seconde in code unit
182  REAL t_ff = SQRT(3.*M_PI/(32.*NEWTON_G * rho_m/ fact_rho));
183  t_ff /= fact_t;
184 
185  // local Jeans time in seconde in code unit
186  REAL t_j = dx/cell->field.a;
187 
188  // star formation rate in kg/s/m3 in code unit
189  REAL SFR = param->stars->efficiency * cell->field.d / t_ff * t_j/t_ff;
190 
191  // Average number of stars created
192  REAL lambda = SFR / mlevel * dt * dv;
193  //printf("rho=%e tff=%e tj=%e SFR=%e\n",cell->field.d, t_ff, t_j, SFR);
194 
195 #endif //SCHAYE
196 
197  int N = gpoiss(lambda); //Poisson drawing
198  //printf("AVG star creation =%e /eff %d\n",lambda,N);
199  REAL M_in_cell = cell->field.d * POW(2.0,-3.0*level); // mass of the curent cell in code unit
200  if(N * mlevel >= M_in_cell) N = 0.9*M_in_cell / mlevel ; // 0.9 to prevent void cells
201 
202  return N;
